67  while (el_it != pref_end(p_nd))

75  std::cerr << std::endl;

79  inode_const_pointer p_internal = static_cast<inode_const_pointer>(p_nd);

84  const size_type num_children = std::distance(p_internal->begin(),
85  p_internal->end());

88  {

90  std::advance(child_it, num_children - child_i - 1);
